Loading Effect

An animated gif generator is availible here.


    #container{
	  padding-left:50px;
	  padding-top:60px;
	}
	
	button{
	  margin-top:60px;
	}
	
	#loader{
	 display:none;
	}
   
     $(document).ready(function()
	  {
	  
	    $("button").click(function()
		 {
		   $(this).fadeOut(300);
		   
		   $("#loader").show();
		   
		   setTimeout(ShowPicture,6000);
		   
		 })
		 

         function ShowPicture()
         {
             $("#loader").html("<img src='kittens.jpg' alt='kittens'/>")      
         }   		 
	 });
   
 
     <div id="container">
  
       <div> id="loader"><img src="loader.gif" alt="loader"/></div>
       <button>Load</button>
  
     </div>
